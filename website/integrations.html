Quick Start

# pip install clsplusplus
from clsplusplus import CLS

client = CLS(api_key="")

# Store a memory
client.memories.encode(content="User prefers dark mode", agent_id="a1")

# Retrieve memories
results = client.memories.retrieve(query="user preferences", agent_id="a1")
for item in results.items:
    print(item.text, item.confidence)
// Write memory
const res = await fetch("https://clsplusplus-api.onrender.com/v1/memories/encode", {
  method: "POST",
  headers: {
    "Content-Type": "application/json",
    "Authorization": "Bearer "
  },
  body: JSON.stringify({ text: "User prefers dark mode", namespace: "default" })
});

// Read memory
const read = await fetch("https://clsplusplus-api.onrender.com/v1/memories/retrieve", {
  method: "POST",
  headers: {
    "Content-Type": "application/json",
    "Authorization": "Bearer "
  },
  body: JSON.stringify({ query: "user preferences", namespace: "default" })
});
# Write memory
curl -X POST https://clsplusplus-api.onrender.com/v1/memories/encode \
  -H "Content-Type: application/json" \
  -H "Authorization: Bearer " \
  -d '{"text": "User prefers dark mode", "namespace": "default"}'

# Read memory
curl -X POST https://clsplusplus-api.onrender.com/v1/memories/retrieve \
  -H "Content-Type: application/json" \
  -H "Authorization: Bearer " \
  -d '{"query": "user preferences", "namespace": "default"}'
